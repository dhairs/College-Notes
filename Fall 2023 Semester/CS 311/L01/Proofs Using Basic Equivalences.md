$\neg(p\wedge q)\vee p$

$\equiv$ \<DeMorgan's>

$(\neg p\vee\neg q)\vee p$

$\equiv$ \<Associative Law>

$\neg p\vee (\neg q\vee p)$

$\equiv$ \<Commutative>

$\neg p\vee(p\vee\neg q)$

$\equiv$ \<Associative>

$(\neg p\vee p)\vee\neg q$

$\equiv$ \<Excluded Middle>

T $\vee\neg q$

$\equiv$ \<Domination>

$T$
